          tensiones = np.array([fila_especifica[f'Bus_{j:04d} tension (kV)'] for j in barras])

          potencia_activa = np.zeros((num_barras, num_barras))
          potencia_reactiva = np.zeros((num_barras, num_barras))
          
          for x in range(len(barras)):
              for y in range(x+1, len(barras)):
                  barra_x = barras[x]
                  barra_y = barras[y]
                  potencia_activa[x, y] = fila_especifica.get(f'P{barra_x}-{barra_y}', 0)
                  potencia_activa[y, x] = fila_especifica.get(f'P{barra_y}-{barra_x}', 0)
                  potencia_reactiva[x, y] = fila_especifica.get(f'Q{barra_x}-{barra_y}', 0)
                  potencia_reactiva[y, x] = fila_especifica.get(f'Q{barra_y}-{barra_x}', 0)
          
          # Crear la imagen (13x13x4) para el escenario
          imagen = np.zeros((num_barras, num_barras, 4))
          
          # Canal R - Tensiones
          for k in range(num_barras):
              imagen[k, k, 0] = tensiones[k]
              
          # Canal G - Ángulos
          for k in range(num_barras):
              imagen[k, k, 1] = angulos[k]
              
          # Canal B - Potencias Activas
          for x in range(num_barras):
              for y in range(num_barras):
                  if x != y:
                      imagen[x, y, 2] = potencia_activa[x, y]
          
          # Canal A - Potencias Reactivas
          for x in range(num_barras):
              for y in range(num_barras):

                      imagen[x, y, 3] = potencia_reactiva[x, y]

      def grad_cam(model, X, target_index):
          """
          Calcula la importancia de cada canal de entrada (tensión, ángulo, P_ij, Q_ij) usando Grad-CAM.
          
          Args:
              model: Modelo CNN entrenado.
              X: Lote de datos de entrada (batch).
              target_index: Índice de la salida objetivo.
              
          Returns:
              importance: Importancia promedio de cada canal.
          """
          with tf.GradientTape() as tape:
              tape.watch(X)  # Vigila las entradas
              predictions = model(X)  # Realiza las predicciones
              target_output = predictions[:, target_index]  # Selecciona la salida objetivo
      ​
          # Calcula los gradientes de la salida objetivo con respecto a las entradas
          gradients = tape.gradient(target_output, X)
      ​
          # Promedia los gradientes absolutos para cada canal
          importance = tf.reduce_mean(tf.abs(gradients), axis=(0, 1, 2))  # Promedio por canal
          return importance.numpy()
